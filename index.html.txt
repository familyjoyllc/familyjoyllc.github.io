        const languageSwitcher = document.getElementById('language-switcher');
        const htmlElement = document.documentElement;

        // Function to set the language and update content
        function setLanguage(lang) {
            htmlElement.lang = lang;
            htmlElement.dir = (lang === 'ar') ? 'rtl' : 'ltr';
            localStorage.setItem('lang', lang);

            // Update button text
            languageSwitcher.textContent = translations[lang].langButton;

            // Update all elements with data-i18n attributes
            document.querySelectorAll('[data-i18n]').forEach(element => {
                const key = element.getAttribute('data-i18n');
                if (translations[lang][key]) {
                    element.innerHTML = translations[lang][key];
                }
            });

            // Specific updates for supplier names that are not data-i18n attributes
            const supplierDivs = document.querySelectorAll('#key-suppliers .card .grid > div');
            const supplierKeys = [
                'supplierZamzam', 'supplierKaram', 'supplierSJ', 'supplierSamaJordan', 'supplierJok',
                'supplierAlHazaa', 'supplierSamaJordanGroup', 'supplierFocid', 'supplierUnicharm',
                'supplierBabyJoy', 'supplierAlyoumBakery', 'supplierHalloumi', 'supplierComma',
                'supplierPackNature', 'supplierArabianDevelopment'
            ];
            supplierDivs.forEach((div, index) => {
                if (supplierKeys[index] && translations[lang][supplierKeys[index]]) {
                    div.textContent = translations[lang][supplierKeys[index]];
                }
            });
        }

        // Initialize language based on localStorage or default to Arabic
        const storedLang = localStorage.getItem('lang') || 'ar';
        setLanguage(storedLang);

        // Add event listener to the language switcher button
        languageSwitcher.addEventListener('click', () => {
            const currentLang = htmlElement.lang;
            const newLang = (currentLang === 'ar') ? 'en' : 'ar';
            setLanguage(newLang);
        });
